```python
class Solution:
    def levelOrder(self, root):

        if not root:
            return []

        q = collections.deque()
        q.append(root)

        result = []

        while q:

            _len = len(q)

            level = []
            for _ in range(_len):

                cur = q.popleft()
                level.append(cur.val)

                if cur.left:
                    q.append(cur.left)
                if cur.right:
                    q.append(cur.right)

            result.append(level)

        return result
```
